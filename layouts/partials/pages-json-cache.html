{{- $url := "https://docs.datadoghq.com/pages.json" -}} {{- $res := resources.GetRemote $url -}} {{- with $res.Err -}} {{- errorf "Failed to fetch remote JSON %s: %s" $url . -}} {{- end -}} {{- $pages := $res | transform.Unmarshal -}} {{- if not $pages -}} {{- errorf "Failed to unmarshal JSON from: %s" $url -}} {{- end -}} {{- return $pages -}}
