{{/* Resolves a single endpoint from a spec version by operationId and computes its region map. Required dict keys: operationids — slice of operationIds to try (one is expected to match this version's spec; the last match wins, preserving original behavior). version — "v1" or "v2"; used to hit the cached operationId index built by partials/api/load-specs.html. generalRegions — fallback region map (from regions.html) when the action has no `servers` override. Returns a dict { pathKey, actionType, action, regions } or nil if no operationId in the slice exists in this version. */}} {{ $s := newScratch }} {{ $operationids := .operationids }} {{ $version := .version }} {{ $generalRegions := .generalRegions }} {{ range $opId := $operationids }} {{ $match := partial "api/load-specs.html" (dict "version" $version "lookup" $opId) }} {{ with $match }} {{ $action := .action }} {{ $s.SetInMap "data" "pathKey" .pathKey }} {{ $s.SetInMap "data" "actionType" .actionType }} {{ $s.SetInMap "data" "action" $action }} {{ $s.Set "regions" dict }} {{ with $action.servers }} {{ $servers := index . 0 }} {{ $serverURL := "" }} {{ if isset $servers.variables.site "enum" }} {{ range $domain := $servers.variables.site.enum }} {{ $serverURL = replace $servers.url "{subdomain}" $servers.variables.subdomain.default }} {{ $serverURL = replace $serverURL "{site}" $domain }} {{ range $allowedRegion := site.Params.allowedRegions }} {{ if eq $allowedRegion.domain $domain }} {{ $s.SetInMap "regions" $allowedRegion.value $serverURL }} {{ end }} {{ end }} {{ end }} {{ if eq (len ($s.Get "regions")) 0 }} {{ range $domain := $servers.variables.site.enum }} {{ $serverURL = replace $servers.url "{site}" $domain }} {{ range $allowedRegion := site.Params.allowedRegions }} {{ if in $allowedRegion.exact_domains $domain }} {{ $s.SetInMap "regions" $allowedRegion.value $serverURL }} {{ end }} {{ end }} {{ end }} {{ end }} {{ else }} {{ $serverURL = replace $servers.url "{site}" $servers.variables.site.default }} {{ $s.SetInMap "regions" "local" $serverURL }} {{ end }} {{ else }} {{ $s.Set "regions" $generalRegions }} {{ end }} {{ $s.SetInMap "data" "regions" ($s.Get "regions") }} {{ end }} {{ end }} {{ return ($s.Get "data") }}
